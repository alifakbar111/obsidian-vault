# `moci list` Command — Implementation Plan

**Repo:** https://github.com/zero-one-group/moci
**Branch:** `feat/list-templates`
**Final PR title:** `feat(cli): add list command to discover monorepo templates`

## Goal

Add a `moci list` subcommand to the `moci` CLI that discovers and displays all available templates from `zero-one-group/monorepo/contents/templates` via the GitHub Contents API, with disk caching for offline use. Fills the biggest DX gap in the ZOG ecosystem: today the only way to discover what templates exist is to visit GitHub manually.

## Why this lands

- **Zero competition** — 0 issues, 0 PRs, 0 forks, 0 releases on `moci`
- **Mirrors the merged pattern** — `moci list` is a toolchain-style CLI addition, same shape as #135 (biome→oxc) and #133 (pre-commit hooks)
- **README's stated promise** — "manage packages, scaffold templates, and run migrations with ease" is currently unfulfilled
- **Natural sibling to `init`** — `init` already uses `giget` to download templates; `list` completes the discovery flow

## Tech choices

| Concern | Choice | Why |
|---|---|---|
| HTTP client | Native `fetch` (Node 22+) | Already used in `utils.ts`, no new dep |
| Cache | `~/.moci/cache.json` | Standard XDG-ish path, no dep |
| Cache invalidation | TTL 1h + `--refresh` flag | Balances freshness + speed |
| Table | Roll our own (~15 lines) | Avoid `cli-table3` dep for one use |
| Tests | `vitest` | Same as the main monorepo uses |
| CI | Add `.github/workflows/ci.yml` | Repo has zero CI currently |

### Task 0.2: Cache utility

**File:** `src/utils/cache.ts`

```ts
import { existsSync, mkdirSync, readFileSync, writeFileSync } from "node:fs";
import { homedir } from "node:os";
import { join } from "node:path";

const CACHE_DIR = process.env.MOCI_CACHE_DIR
  ?? join(process.env.XDG_CACHE_HOME ?? join(homedir(), ".cache"), "moci");

const CACHE_TTL_MS = 60 * 60 * 1000; // 1 hour

export interface CacheEntry<T> {
  data: T;
  cachedAt: number; // epoch ms
}

function ensureDir(): void {
  mkdirSync(CACHE_DIR, { recursive: true });
}

function pathFor(key: string): string {
  // Replace path separators so sub-keys like "list:react:monorepo" work
  return join(CACHE_DIR, `${key.replace(/[\\/]/g, "_")}.json`);
}

export function cacheGet<T>(key: string): T | null {
  if (!existsSync(CACHE_DIR)) return null;
  const file = pathFor(key);
  if (!existsSync(file)) return null;
  try {
    const raw = readFileSync(file, "utf8");
    const entry = JSON.parse(raw) as CacheEntry<T>;
    if (Date.now() - entry.cachedAt > CACHE_TTL_MS) return null;
    return entry.data;
  } catch {
    return null;
  }
}

export function cachePut<T>(key: string, data: T): void {
  ensureDir();
  const entry: CacheEntry<T> = { data, cachedAt: Date.now() };
  writeFileSync(pathFor(key), JSON.stringify(entry), "utf8");
}

export function cacheClear(key?: string): void {
  if (!existsSync(CACHE_DIR)) return;
  if (key) {
    const file = pathFor(key);
    if (existsSync(file)) writeFileSync(file, "", "utf8");
    return;
  }
  for (const f of readdirSync(CACHE_DIR).filter((f) => f.endsWith(".json"))) {
    writeFileSync(join(CACHE_DIR, f), "", "utf8");
  }
}

export { CACHE_DIR };
```

  it("returns null on cache miss", () => {
    expect(cache.cacheGet("missing")).toBeNull();
  });

  it("returns the value on cache hit", () => {
    cache.cachePut("k", { hello: "world" });
    expect(cache.cacheGet("hello")).toBeNull();
    expect(cache.cacheGet("k")).toEqual({ hello: "world" });
  });

    // Manually write an expired entry
    cache.cachePut("expired", { v: 1 });
    const file = join(testDir, "expired.json");
    const entry = JSON.parse(require("node:fs").readFileSync(file, "utf8"));
    entry.cachedAt = Date.now() - 2 * 60 * 60 * 1000; // 2 hours ago
    require("node:fs").writeFileSync(file, JSON.stringify(entry), "utf8");
    expect(cache.cacheGet("expired")).toBeNull();
  });

  it("clear() empties a specific key", () => {
    cache.cachePut("a", 1);
    cache.cachePut("b", 2);
    cache.cacheClear("a");
    expect(cache.cacheGet("a")).toBeNull();
    expect(cache.cacheGet("b")).toBe(2);
  });

  it("clear() with no arg empties all keys", () => {
    cache.cachePut("a", 1);
    cache.cachePut("b", 2);
    cache.cacheClear();
    expect(cache.cacheGet("a")).toBeNull();
    expect(cache.cacheGet("b")).toBeNull();

### Task 0.4: Table renderer

**File:** `src/utils/table.ts`

```ts
export interface TableColumn {
  header: string;
  width?: number; // computed if omitted
  align?: "left" | "right";
}

export function renderTable(
  columns: TableColumn[],
  rows: string[][],
): string {
  // Compute column widths from header + all row values
  const widths = columns.map((col, idx) => {
    const cellMax = rows.reduce(
      (max, row) => Math.max(max, (row[idx] ?? "").length),
      0,
    );
    return Math.max(col.width ?? 0, col.header.length, cellMax);
  });

  const pad = (s: string, w: number, align: "left" | "right" = "left") =>
    align === "right" ? s.padStart(w) : s.padEnd(w);

  const sep = "─";
  const top = `┌${widths.map((w) => sep.repeat(w + 2)).join("┬")}┐`;
  const mid = `├${widths.map((w) => sep.repeat(w + 2)).join("┼")}┤`;
  const bot = `└${widths.map((w) => sep.repeat(w + 2)).join("┴")}┘`;

  const renderRow = (cells: string[]) =>
    `│ ${cells
      .map((c, i) => pad(c, widths[i], columns[i].align))
      .join(" │ ")} │`;

  const lines = [
    top,
    renderRow(columns.map((c) => c.header)),
    mid,
    ...rows.map((r) => renderRow(r)),
    bot,
  ];

  return lines.join("\n");
}
```

### Task 0.6: Templates fetcher

**File:** `src/utils/templates.ts`

```ts
import { cacheGet, cachePut } from "./cache";

export interface TemplateInfo {
  name: string;
  framework: string;
  frameworkVersion: string;
  language: string;
  lastUpdated: string;
  url: string;
  sizeKb: number;
}

const GITHUB_API = "https://api.github.com";

interface GhContent {
  name: string;
  path: string;
  type: "file" | "dir";
  sha: string;
  size?: number;
  html_url: string;
}

interface GhCommit {
  sha: string;
  commit: { author: { date: string } };
}

interface GhFileContent {
  content: string;
  encoding: string;
}

interface PkgJson {
  dependencies?: Record<string, string>;
  devDependencies?: Record<string, string>;
}

async function gh<T>(path: string): Promise<T> {
  const res = await fetch(`${GITHUB_API}${path}`, {
    headers: {
      Accept: "application/vnd.github.v3+json",
      "User-Agent": "moci-cli",
    },
  });
  if (!res.ok) {
    throw new Error(
      `GitHub API ${res.status} ${res.statusText} for ${path}`,
    );
  }
  return (await res.json()) as T;
}

async function listTemplateDirs(
  org: string,
  repo: string,
  path: string,
): Promise<GhContent[]> {
  const items = await gh<GhContent[]>(
    `/repos/${org}/${repo}/contents/${path}`,
  );
  return items.filter((i) => i.type === "dir");
}

async function lastCommitDate(
  org: string,
  repo: string,
  path: string,
): Promise<string> {
  try {
    const commits = await gh<GhCommit[]>(
      `/repos/${org}/${repo}/commits?path=${encodeURIComponent(path)}&per_page=1`,
    );
    return commits[0]?.commit.author.date ?? new Date(0).toISOString();
  } catch {
    return new Date(0).toISOString();
  }
}

function detectFramework(pkg: PkgJson): {
  framework: string;
  frameworkVersion: string;
  language: string;
} {
  const all = { ...(pkg.dependencies ?? {}), ...(pkg.devDependencies ?? {}) };
  const pick = (key: string, framework: string, lang = "typescript") => {
    if (all[key]) {
      return { framework, frameworkVersion: all[key]!, language: lang };
    }
    return null;
  };
  return (
    pick("next", "next") ??
    pick("@remix-run/react", "remix") ??
    pick("@tanstack/react-start", "tanstack-start") ??
    pick("@tanstack/react-router", "tanstack-router") ??
    pick("astro", "astro") ??
    pick("nuxt", "nuxt") ??
    pick("vue", "vue") ??
    pick("expo", "expo") ??
    pick("react", "react") ??
    pick("fastapi", "fastapi", "python") ??
    { framework: "unknown", frameworkVersion: "0", language: "unknown" }
  );
}

async function readTemplateMeta(
  org: string,
  repo: string,
  templatePath: string,
): Promise<{ framework: string; frameworkVersion: string; language: string }> {
  try {
    const file = await gh<GhFileContent>(
      `/repos/${org}/${repo}/contents/${templatePath}/package.json`,
    );
    if (file.encoding === "base64" && file.content) {
      const decoded = JSON.parse(
        Buffer.from(file.content, "base64").toString("utf8"),
      ) as PkgJson;
      return detectFramework(decoded);
    }
  } catch {
    // Fall through to unknown
  }
  return { framework: "unknown", frameworkVersion: "0", language: "unknown" };
}

export interface ListOptions {
  org: string;
  repo: string;
  path: string;
  useCache: boolean;
  refresh: boolean;
}

export async function listTemplates(
  opts: ListOptions,
): Promise<TemplateInfo[]> {
  const cacheKey = `list:${opts.org}:${opts.repo}:${opts.path}`;

  if (opts.useCache && !opts.refresh) {
    const cached = cacheGet<TemplateInfo[]>(cacheKey);
    if (cached) return cached;
  }

  const dirs = await listTemplateDirs(opts.org, opts.repo, opts.path);

  // Fetch metadata in parallel (with concurrency limit)
  const CONCURRENCY = 4;
  const results: TemplateInfo[] = [];
  for (let i = 0; i < dirs.length; i += CONCURRENCY) {
    const batch = dirs.slice(i, i + CONCURRENCY);
    const batchResults = await Promise.all(
      batch.map(async (dir) => {
        const templatePath = `${opts.path}/${dir.name}`;
        const [meta, lastUpdated] = await Promise.all([
          readTemplateMeta(opts.org, opts.repo, templatePath),
          lastCommitDate(opts.org, opts.repo, templatePath),
        ]);
        return {
          name: dir.name,
          ...meta,
          lastUpdated,
          url: `https://github.com/${opts.org}/${opts.repo}/tree/main/${templatePath}`,
          sizeKb: Math.round((dir.size ?? 0) / 1024),
        };
      }),
    );
    results.push(...batchResults);
  }

  results.sort((a, b) => a.name.localeCompare(b.name));

  if (opts.useCache) {
    cachePut(cacheKey, results);
  }

  return results;
}
```

import { defineCommand } from "citty";
import { consola } from "consola";
import { listTemplates, type TemplateInfo } from "../utils/templates";
import { renderTable } from "../utils/table";

const RELATIVE_TIME_FORMAT = new Intl.RelativeTimeFormat("en", { numeric: "auto" });

function timeAgo(iso: string): string {
  const then = new Date(iso).getTime();
  if (!then) return "unknown";
  const diffMs = Date.now() - then;
  const diffSec = Math.floor(diffMs / 1000);
  if (diffSec < 60) return RELATIVE_TIME_FORMAT.format(-diffSec, "second");
  const diffMin = Math.floor(diffSec / 60);
  if (diffMin < 60) return RELATIVE_TIME_FORMAT.format(-diffMin, "minute");
  const diffHr = Math.floor(diffMin / 60);
  if (diffHr < 24) return RELATIVE_TIME_FORMAT.format(-diffHr, "hour");
  const diffDay = Math.floor(diffHr / 24);
  if (diffDay < 30) return RELATIVE_TIME_FORMAT.format(-diffDay, "day");
  const diffMonth = Math.floor(diffDay / 30);
  if (diffMonth < 12) return RELATIVE_TIME_FORMAT.format(-diffMonth, "month");
  return RELATIVE_TIME_FORMAT.format(-Math.floor(diffDay / 365), "year");
}

function cleanVersion(v: string): string {
  // "19.2.4" -> "19"; "3.5.13" -> "3.5"; "^3.5.13" -> "3.5"
  const stripped = v.replace(/^[\^~]/, "");
  const parts = stripped.split(".");
  if (parts.length >= 2) return `${parts[0]}.${parts[1]}`;
  return parts[0] ?? "0";
}

function formatTemplate(t: TemplateInfo): string[] {
  return [
    t.name,
    `${t.framework}@${cleanVersion(t.frameworkVersion)}`,
    t.language,
    timeAgo(t.lastUpdated),
  ];
}

export default defineCommand({
  meta: {
    name: "list",
    description: "List available monorepo templates",
  },
  args: {
    org: {
      type: "string",
      description: "GitHub organization",
      default: "zero-one-group",
    },
    repo: {
      type: "string",
      description: "GitHub repository",
      default: "monorepo",
    },
    path: {
      type: "string",
      description: "Path within the repo to scan for templates",
      default: "templates",
    },
    json: {
      type: "boolean",
      description: "Output as JSON",
      default: false,
    },
    refresh: {
      type: "boolean",
      description: "Bypass cache and fetch fresh data",
      default: false,
    },
    "no-cache": {
      type: "boolean",
      description: "Don't read or write the cache for this call",
      default: false,
    },
    help: {
      type: "boolean",
      description: "Print help and information about the command",
      default: false,
      alias: "h",
    },
  },
  async run({ args }) {
    try {
      const useCache = !args["no-cache"];
      const templates = await listTemplates({
        org: args.org,
        repo: args.repo,
        path: args.path,
        useCache,
        refresh: args.refresh,
      });

      if (args.json) {
        consola.log(JSON.stringify(templates, null, 2));
        return;
      }

      if (templates.length === 0) {
        consola.warn(
          `No templates found in ${args.org}/${args.repo}/${args.path}`,
        );
        return;
      }

      const table = renderTable(
        [
          { header: "Template" },
          { header: "Framework" },
          { header: "Language" },
          { header: "Last Update" },
        ],
        templates.map(formatTemplate),
      );

      consola.log(table);
      consola.log(
        `\n${templates.length} template${templates.length === 1 ? "" : "s"} available. Run \`moci info <name>\` for details.`,
      );
    } catch (error) {
      consola.error(
        `Failed to list templates: ${error instanceof Error ? error.message : String(error)}`,
      );
      process.exit(1);
    }
  },
});
